import java.util.*;
class Solution {
    public int solution(int[][] board, int[] moves) {
        int answer = 0;

        Stack<Integer> stack = new Stack<>();
        stack.push(0);

        for (int i = 0; i < moves.length; i++)
        {
            for (int j = 0; j < board.length; j++)
            {
                if (board[j][moves[i] - 1] != 0)
